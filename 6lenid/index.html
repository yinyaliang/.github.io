    6位唯一码

    package main

import (
        "fmt"
        "math/rand"
        "time"
)

func un()string{
    str := "abcdefghijklmnopqrstuvwxyz1234567890" // 取值池
    timestamp := time.Now().UnixNano()  // 纳秒
    un := make([]byte,6)
    rand.Seed(timestamp)
    for i := range un {
        un[i] = str[rand.Intn(len(str))]
    }
    return string(un)
}

func main(){
    testMap := make(map[string]int)
    testRange := 2

    for i:=1;i <= testRange;i++{
        un := un()
        _ ,ok := testMap[un]
        if ok{
            testMap[un] += 1
        }else{
            testMap[un] = 0
        }
    }

    conflict := 0
    for _,val := range testMap{
        if val != 0{
            conflict += 1
        }
    }
    fmt.Println(testMap)
    fmt.Println(conflict)
}
